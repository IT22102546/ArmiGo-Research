// ============================================================================
// CLEANUP FUNCTION
// ============================================================================

async function cleanupExistingData(): Promise<void> {
  log.section("CLEANUP: Removing existing seed data");

  // Delete in reverse dependency order to avoid foreign key violations
  // Order matters: child tables first, then parent tables

  // Logging and Feedback
  await prisma.feedback.deleteMany({});
  await prisma.auditLog.deleteMany({});

  // Authentication and Security
  await prisma.passwordReset.deleteMany({});
  await prisma.emailVerification.deleteMany({});
  await prisma.deviceToken.deleteMany({});
  await prisma.refreshToken.deleteMany({});
  await prisma.authSession.deleteMany({});
  await prisma.accountLockout.deleteMany({});

  // Access and Availability
  await prisma.temporaryAccess.deleteMany({});
  await prisma.teacherAvailability.deleteMany({});

  // Progress and Communication
  await prisma.studentProgress.deleteMany({});
  await prisma.chatMessage.deleteMany({});
  await prisma.notification.deleteMany({});
  await prisma.scheduledNotification.deleteMany({});

  // Seminars
  await prisma.seminarRegistration.deleteMany({});
  await prisma.seminar.deleteMany({});

  // Publications
  await prisma.publicationReview.deleteMany({});
  await prisma.publicationPurchase.deleteMany({});
  await prisma.publicationMedium.deleteMany({});
  await prisma.publicationSubject.deleteMany({});
  await prisma.publicationGrade.deleteMany({});
  await prisma.publication.deleteMany({});

  // Payments and Wallet
  await prisma.payment.deleteMany({});
  await prisma.walletTransaction.deleteMany({});
  await prisma.wallet.deleteMany({});

  // Exams
  await prisma.examAnswer.deleteMany({});
  await prisma.examAttempt.deleteMany({});
  await prisma.examRanking.deleteMany({});
  await prisma.examQuestion.deleteMany({});
  await prisma.questionGroup.deleteMany({});
  await prisma.examSection.deleteMany({});
  await prisma.exam.deleteMany({});
  await prisma.examTemplate.deleteMany({});

  // Classes and Sessions
  await prisma.attendance.deleteMany({});
  await prisma.enrollment.deleteMany({});
  await prisma.sessionParticipant.deleteMany({});
  await prisma.recording.deleteMany({});
  await prisma.classSession.deleteMany({});
  await prisma.videoSession.deleteMany({});
  await prisma.timetableChange.deleteMany({});
  await prisma.timetable.deleteMany({});

  // Course Materials
  await prisma.courseMaterial.deleteMany({});

  // Classes
  await prisma.class.deleteMany({});

  // Subject structure
  await prisma.gradeSubjectOffering.deleteMany({});

  // User-related data
  await prisma.faceRecognition.deleteMany({});
  await prisma.studentStarAward.deleteMany({});
  await prisma.studentSubject.deleteMany({});
  await prisma.studentProfile.deleteMany({});
  await prisma.teacherSubjectAssignment.deleteMany({});
  await prisma.teacherProfile.deleteMany({});
  await prisma.user.deleteMany({});

  // Sibling groups
  await prisma.siblingGroup.deleteMany({});

  log.success("Cleanup completed");
}
